routine fibonacciRecursive(n: integer): integer is
    if n <= 1 then
        return 1
    else
        return fibonacciRecursive(n - 1) + fibonacciRecursive(n - 2)
    end
end

routine fibonacciIterative(n: integer): integer is
    if n <= 1 then
        return 1
    end
    var a is 1;
    var b is 1;
    for i in 2 .. n loop
        var c is a + b;
        a := b;
        b := c
    end
    return b
end

routine fibonacci(n: integer, isIterative: boolean) is
    var result: integer
    if isIterative then
        result := fibonacciIterative(n)
    else
        result := fibonacciRecursive(n)
    end
    print result
end
