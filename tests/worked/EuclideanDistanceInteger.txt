var result : integer

type Point is record
    var x: integer
    var y: integer
end

routine squareRoot(a: integer, sqrt: array[1] integer) is
    var low : integer
    var high : integer
    low := 0
    high := 10000
    while low + 1 < high loop
        var mid : integer
        mid := (low + high) / 2
        if mid * mid <= a then
            low := mid
        else
            high := mid
        end
    end
    sqrt[1] := low
end

routine euclideanDistance(p1x: integer, p1y: integer, p2x: integer, p2y: integer) is
    var p1 : Point
    var p2 : Point
    p1.x := p1x
    p1.y := p1y
    p2.x := p2x
    p2.y := p2y

    var sqrt : array [1] integer
    squareRoot((p1.x - p2.x) * (p1.x - p2.x) + (p1.y - p2.y) * (p1.y - p2.y), sqrt)
    result := sqrt[1]
end
